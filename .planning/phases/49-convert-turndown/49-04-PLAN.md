    describe('sanitizeHtml', () => {
      it.each([
        // [description, input fragment, substring that MUST be absent from output]
        ['<script> subtree stripped',
          '<p>ok</p><script>alert(1)</script>',
          'alert(1)'],
        ['<style> subtree stripped',
          '<style>.x{color:red}</style><p>ok</p>',

        ['<noscript> subtree stripped',
          '<noscript>fallback</noscript><p>ok</p>',
          'fallback'],
        ['[aria-hidden="true"] subtree stripped',
          '<div aria-hidden="true">hidden-icon</div><p>visible</p>',
          'hidden-icon'],
      ])('strips %s', (_desc, input, forbidden) => {
        const output = sanitizeHtml(input)
        expect(output).not.toContain(forbidden)
        expect(output).toContain('ok') // or 'visible' for the aria case — adjust per row
      })

      it('strips data-v-* attributes but preserves class/href/alt', () => {
        const input = '<p data-v-abc123 class="x" data-other="keep"><a href="/h" data-v-xyz789>link</a></p>'
        const output = sanitizeHtml(input)
        expect(output).not.toMatch(/data-v-/)
        expect(output).toContain('class="x"')
        expect(output).toContain('data-other="keep"')
        expect(output).toContain('href="/h"')
      })

      it('is deterministic — same input yields byte-equal output', () => {
        const input = '<h1>T</h1><p data-v-a class="k">body <span class="badge">X</span></p>'
        expect(sanitizeHtml(input)).toBe(sanitizeHtml(input))
      })
    })
