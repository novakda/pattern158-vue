// node_modules/turndown/types/turndown.d.ts
declare class TurndownService {
  constructor(options?: TurndownService.Options)
  addRule(key: string, rule: TurndownService.Rule): this
  use(plugin: TurndownService.Plugin | TurndownService.Plugin[]): this
  turndown(input: string | TurndownService.Node): string
}

namespace TurndownService {
  interface Options {
    headingStyle?: 'setext' | 'atx'
    bulletListMarker?: '-' | '+' | '*'
    codeBlockStyle?: 'indented' | 'fenced'
    emDelimiter?: '_' | '*'
    strongDelimiter?: '__' | '**'
    linkStyle?: 'inlined' | 'referenced'
    linkReferenceStyle?: 'full' | 'collapsed' | 'shortcut'
    // ... plus others not used by this plan
  }

  interface Rule {
    filter: string | string[] | FilterFunction
    replacement: ReplacementFunction
  }

  type FilterFunction = (node: Node, options: Options) => boolean
  type ReplacementFunction = (content: string, node: Node, options: Options) => string

  type Plugin = (service: TurndownService) => void
}
```

       /**
        * configureTurndown — factory for a configured TurndownService. Returns a fresh
        * instance each call (callers own the lifecycle; GFM plugin registration is
        * idempotent per-instance). Custom rules register BEFORE .use(gfm) so the
        * GFM plugin's own rules don't shadow ours.
        *
        * CONV-01: Turndown 7.2.4 + @joplin/turndown-plugin-gfm full plugin.
        * CONV-03: images emit alt text only — no ![]() Markdown, no base64 data URLs.
        * CONV-05: badge/pill/chip/tag/severity/category spans render as **bold**.
        * CONV-06: DOM-order preservation is Turndown's default — verified by tests.
        * CONV-07: hrefs preserved verbatim via linkStyle: 'inlined'.
        */
       export function configureTurndown(): TurndownService {
         const service = new TurndownService({
           headingStyle: 'atx',
           bulletListMarker: '-',
           codeBlockStyle: 'fenced',
           emDelimiter: '*',
           strongDelimiter: '**',
           linkStyle: 'inlined',
           linkReferenceStyle: 'full',
         })

         // CONV-03: alt-text-only image rule. Override Turndown's default image
         // handler so ![]( ) Markdown and base64 data URLs never appear in output.
         service.addRule('image-alt-only', {
           filter: 'img',
           replacement: (_content, node) => {
             const alt = (node as Element).getAttribute('alt') ?? ''
             return alt.length === 0 ? '' : alt
           },
         })

         // CONV-05: pattern158 design-system badge/pill passthrough as **bold**.
         // Class allowlist matches Phase 5+/7+ design tokens: badge, pill, chip,
         // tag, tag-*, severity-*, category-*. Nested icon subtrees flatten to
         // text-only via Turndown's natural content recursion.
         service.addRule('pattern158-badges', {
           filter: (node) => {
             const tag = node.nodeName.toLowerCase()
             if (tag !== 'span' && tag !== 'a') return false
             const cls = (node as Element).getAttribute('class') ?? ''
             return /(^|\s)(badge|pill|chip|tag|tag-\w+|severity-\w+|category-\w+)(\s|$)/.test(cls)
           },
           replacement: (content) => {
             const trimmed = content.trim()
             return trimmed.length === 0 ? '' : `**${trimmed}**`
           },
         })

         // CONV-01: GFM plugin AFTER custom rules — so tables/strikethrough/task-list
         // rules from the plugin don't shadow the img + badge rules above.
         service.use(gfm)

         return service
       }
       ```

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-49-06 | Denial of Service | pattern158-badges rule — ReDoS via crafted class attribute | accept/mitigate | Regex `/(^|\s)(badge|pill|chip|tag|tag-\w+|severity-\w+|category-\w+)(\s|$)/` has only `\w+` unbounded quantifiers. `\w+` is linear-time (no nested quantifiers, no backtracking explosion). | T-49-07 | Tampering | image-alt-only rule — alt text injection via attacker-controlled alt | mitigate | Sanitizer (Plan 49-01) runs BEFORE configureTurndown's rules fire. Raw `<script>` in alt (`alt="\" onerror=...`) cannot survive into Turndown because sanitizeHtml strips `<script>` subtrees; attribute values are NOT parsed as HTML by Turndown default rules. The alt text emitted as bare Markdown text will be escaped by Turndown's built-in Markdown escapes (*, _, \[ etc.). No injection path to Markdown output. |
| T-49-08 | Information Disclosure | image src leakage via base64 data URLs | mitigate | Rule's `replacement` never reads `src`. Only `alt` is consumed; `src` is effectively dropped. Base64 payloads never reach the Markdown output even if present on the input element. |
| T-49-09 | Tampering | badge rule — HTML content injection via badge inner HTML | mitigate | `content` parameter of `replacement` is the Turndown-processed (Markdown-escaped) content of the element's subtree, NOT raw HTML.
