```ts
export async function captureRoutes(
  config: EditorialConfig,
  routes: readonly Route[],
): Promise<readonly CapturedPage[]> {
  // Precondition: ensure screenshots dir exists once before loop
  await ensureScreenshotDir(config)

  // Load faq.json item count once ‚Äî used on the /faq route only
  const faqItemCount = await loadFaqItemCount(config.exhibitsJsonPath)

  const browser = await launchBrowser(config)
  try {
    const context = await browser.newContext(buildContextOptions())
    try {
      const captured: CapturedPage[] = []
      for (let i = 0; i < routes.length; i += 1) {
        const route = routes[i]
        // capturePage throws CaptureError with route context on any per-route failure.
        // CONTEXT.md "no retries" ethos: we do NOT catch here ‚Äî the run aborts.
        const page = await capturePage(context, config, route, i, faqItemCount)
        captured.push(page)

        // Inter-request delay (CONTEXT.md locked: 1500ms, deterministic).
        // Skip delay after the LAST route ‚Äî no next route to protect against.
        if (i < routes.length - 1) {
          const tempPage = await context.newPage()
          try {
            await tempPage.waitForTimeout(1500)
          } finally {
            await tempPage.close()
          }
        }
      }
      return captured
    } finally {
      await context.close()
    }
  } finally {
    await browser.close()
  }
}
```

    ```ts
    export async function captureRoutes(
      config: EditorialConfig,
      routes: readonly Route[],
    ): Promise<readonly CapturedPage[]> {
      // Fail-loud preflight BEFORE browser boot (CONTEXT.md: "fail early" ethos).
      await ensureScreenshotDir(config)
      const faqItemCount = await loadFaqItemCount(config.exhibitsJsonPath)

      const browser = await launchBrowser(config)
      try {
        const context = await browser.newContext(buildContextOptions())
        try {
          const captured: CapturedPage[] = []
          for (let i = 0; i < routes.length; i += 1) {
            const route = routes[i]
            try {
              const page = await capturePage(context, config, route, i, faqItemCount)
              captured.push(page)
            } catch (err) {
              // Attach route context to non-CaptureError failures so Plan 50's
              // index.ts error printer always has a route to display.
              if (err instanceof CaptureError) {
                throw err
              }
              const message = err instanceof Error ? err.message : String(err)
              throw new CaptureError(
                `Capture failed for ${route.path}: ${message}`,
                { route, cause: err },
              )
            }

            // Inter-request delay (CONTEXT.md locked: 1500ms). Skip after last route.
            if (i < routes.length - 1) {
              const tempPage = await context.newPage()
              try {
                await tempPage.waitForTimeout(1500)
              } finally {
                await tempPage.close()
              }
            }
          }
          return captured
        } finally {
          await context.close()
        }
      } finally {
        await browser.close()
      }
    }
    ```
