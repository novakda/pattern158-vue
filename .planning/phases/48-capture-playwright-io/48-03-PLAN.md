/**
 * capturePage — single-route capture (CAPT-04, CAPT-05, CAPT-06, CAPT-09, CAPT-14, CAPT-15).
 *
 * High-level flow:
 *   1. Create fresh page via context.newPage() (CONTEXT.md lifecycle).
 *   2. Attach console + pageerror listeners (CAPT-14). Collect errors into string[].
 *   3. goto(buildCaptureUrl(baseUrl, route), { timeout: 30_000, waitUntil: 'domcontentloaded' }).
 *   4. Capture HTTP status from response (CAPT-05).
 *   5. Read cf-cache-status response header if present (CAPT-10 observability).
 *   6. waitForSelector('#main-content', { timeout: 10_000 }) — page-ready signal (CAPT-04).
 *   7. If route.path === '/faq': runFaqPreCaptureHooks(page, faqItemCount) (CAPT-07/08).
 *   8. Read page.title() and meta[name="description"] content (CAPT-15).
 *   9. Read mainHtml from page.locator('main#main-content').innerHTML() (CAPT-06).
 *  10. Run detectInterstitial({ title, bodyBytes: mainHtml.length, html: mainHtml }) — non-null aborts (CAPT-11).
 *  11. If route.category === 'exhibit': assert page.locator('.exhibit-detail-title').count() === 1 (CAPT-09).
 *  12. Take screenshot via captureScreenshot(page, buildScreenshotPath(config, index, route)) (CAPT-13).
 *  13. Wait 1500ms (CONTEXT.md locked inter-request delay) — NOTE: delay happens AFTER this function returns,
 *      in the caller (captureRoutes) so per-page function stays pure about its own work. DO NOT put the delay here.
 *  14. Return the CapturedPage. Close the page in a finally block.
 *
 * On any thrown error inside this function, attach the route via CaptureError opts.route
 * if it isn't already attached, and re-throw. Caller uses the route on CaptureError for
 * diagnostic output in Plan 48-06.
 */
export async function capturePage(
  context: BrowserContext,
  config: EditorialConfig,
  route: Route,
  index: number,
  faqItemCount: number,
): Promise<CapturedPage>
```

<!-- Error-wrapping rule (per CONTEXT.md "Error Surface"): -->
- If an underlying error is already a `CaptureError` with `.route` set, re-throw unchanged.
- If it's a `CaptureError` without `.route`, re-throw as a new `CaptureError(err.message, { route, cause: err.cause })`.
- If it's anything else (Playwright TimeoutError, native Error), wrap as `CaptureError(\`Capture failed for ${route.path}: ${err.message}\`, { route, cause: err })`.

<!-- SCAF-08 constraints: -->
- Sequential iteration inside capturePage — no Promise.all over any list.
- No Date.now() / new Date() — the 1500ms delay is done via Playwright's `page.waitForTimeout(1500)` (not a Node setTimeout/Date check). Actually PER CONTEXT.md, the inter-request delay lives in captureRoutes (Plan 48-06), not capturePage.
- No @/ aliases; no os.EOL.

           })

               `Cloudflare bot interstitial detected on ${route.path} — ${interstitialReason}`,
               { route },
             )
           }

               .locator('.exhibit-detail-title')
               .count()

               throw new CaptureError(
                 `Exhibit route ${route.path} did not render .exhibit-detail-title (count=${exhibitTitleCount}) — likely silent 404`,
                 { route },
               )
             }
           }

           // Screenshot (CAPT-13) — full-page PNG at the deterministic filename.
           const screenshotPath = buildScreenshotPath(config, index, route)
           await captureScreenshot(page, screenshotPath)

           const result: CapturedPage = {
             route,
             httpStatus,
             mainHtml,
             title: pageTitle,
             description,
             consoleErrors: [...consoleErrors],
             screenshotPath,
             cfCacheStatus,
           }
           return result
    ```
