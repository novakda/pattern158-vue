    ```ts
    function buildToolVersion(): string {
      try {
        const shortSha = execSync('git rev-parse --short HEAD', { encoding: 'utf8' }).trim()
        const porcelain = execSync('git status --porcelain', { encoding: 'utf8' }).trim()
        const isDirty = porcelain.length > 0
        return `editorial-capture/${shortSha}${isDirty ? '+dirty' : ''}`
      } catch {
        return 'editorial-capture/unknown'
      }
    }
