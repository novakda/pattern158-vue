<objective>
Ship three parallel-safe extractors for the three simplest static pages in the v7.0 markdown-export pipeline: **technologies**, **faq**, and **contact**. Each page has a minimal shape (hero + at most one or two additional sections), so all three fit in a single Wave 2 plan without exceeding context budget.

Each extractor:
1. Imports the same `src/content/*.ts` module the corresponding Vue SFC imports (D-21, single source of truth — LOAD-01 compatible).
2. Composes a `PageDoc` using Phase 38 primitives (`heading`, `paragraph`, `blockquote`) — **never** emits an H1 in `body` (D-11, D-19c/e).
3. Is validated by a co-located `.test.ts` that asserts (a) meta round-trip, (b) first-body-node-is-H2, (c) section coverage, (d) no-H1-anywhere, and (e) a single `toMatchInlineSnapshot()` on the full `PageDoc` (D-19, D-20).

This plan also wires the three new extractor references into `scripts/markdown-export/site-map.ts`, replacing the `notImplemented('/<route>')` stubs shipped by Plan 039-01.

**Mapping rules (per D-12/D-13/D-14/D-15):**
- Module's `heading: string` → `heading(2, heading)` DocNode.
- Module's `intro: string` / `introParagraph: string` / prose text → `paragraph(text)` after the H2.
- `lookingFor` / `criteria` / `items` / `rows` (term+description shape) → `paragraph(strong(label) + text(' — ') + text(description))` wrapped in `list({ ordered: false, items: [...paragraphs] })` per D-14.
- Bare string arrays (e.g. `notLookingFor: string[]`) → `list(stringArray)` (unordered, string sugar per Phase 38 list primitive).
- `links[]` on contactMethods (has `label`+`display`+`href`) → `list({ ordered: false, items: [paragraph([strong(label), text(' — '), link(href, display)])] })`.
- `quote` objects (text + cite) → `blockquote([paragraph(text), paragraph(emphasis(cite))])` per D-15. If `context` is present, append `paragraph(emphasis(context))` as a third child of the blockquote. If `cite` is absent (some home/faq quotes have none), skip the cite paragraph.

```ts
// scripts/markdown-export/extractors/faq.ts
//
// Phase 39 extractor for /faq. Composes the FAQ page shell (hero + colleague
// quotes) from src/content/faqPage.ts. The actual FAQ question/answer items
// live in src/data/faq.ts (JSON) — they are OUT OF SCOPE for Phase 39
// (pure-static extraction of content modules only).
//
// Locked decisions (039-CONTEXT.md):
//   D-08, D-11, D-12, D-15 (quotes → blockquote), D-21, D-23.

import type { DocNode, InlineSpan, PageDoc } from '../ir/types.js'
import type { ExtractorContext } from '../site-map.types.js'
import { FAQ_META } from '../meta.js'
import { heading } from '../primitives/heading.js'
import { paragraph } from '../primitives/paragraph.js'
import { blockquote } from '../primitives/blockquote.js'
import { text } from '../primitives/text.js'
import { hero, colleagueQuotesHeading, colleagueQuotes } from '../../../src/content/faqPage.js'

export function extract(ctx: ExtractorContext): PageDoc {
  const body: DocNode[] = []
  // Lead paragraph from hero.subtitle (hero.title is the PageDoc.title via
  // FAQ_META — not re-emitted in body per D-11).
  body.push(paragraph(hero.subtitle))

  // Colleague quotes section — named export with a heading string → H2.
  body.push(heading(2, colleagueQuotesHeading))
  for (const q of colleagueQuotes) {
    const children: DocNode[] = [paragraph(q.quote)]
    if (q.cite !== undefined) {
      const citeSpans: InlineSpan[] = [
        { kind: 'emphasis', children: [text(q.cite)] },
      ]
      children.push(paragraph(citeSpans))
    }
    if (q.context !== undefined) {
      const ctxSpans: InlineSpan[] = [
        { kind: 'emphasis', children: [text(q.context)] },
      ]
      children.push(paragraph(ctxSpans))
    }
    body.push(blockquote(children))
  }

  return {
    ...FAQ_META,
    sourceRoute: ctx.sourceRoute,
    body,
  }
}
```

```ts
// scripts/markdown-export/extractors/contact.ts
//
// Phase 39 extractor for /contact. Composes PageDoc from src/content/contact.ts
// + five sub-modules under src/content/sections/ (the same set ContactPage.vue
// imports — D-21 single source of truth).
//
// Locked decisions (039-CONTEXT.md):
//   D-08, D-11, D-12, D-13 (no fabricated H3), D-14 (LabeledBullet/DefinitionListItem → bullet list),
//   D-15 (quotes → blockquote), D-21, D-23.

import type { DocNode, InlineSpan, ListItem, PageDoc, StrongSpan } from '../ir/types.js'
import type { ExtractorContext } from '../site-map.types.js'
import { CONTACT_META } from '../meta.js'
import { heading } from '../primitives/heading.js'
import { paragraph } from '../primitives/paragraph.js'
import { list } from '../primitives/list.js'
import { blockquote } from '../primitives/blockquote.js'
import { link } from '../primitives/link.js'
import { text } from '../primitives/text.js'
import { hero, colleagueQuotes } from '../../../src/content/contact.js'
import { roleFit } from '../../../src/content/sections/roleFit.js'
import { companyFit } from '../../../src/content/sections/companyFit.js'
import { cultureFit } from '../../../src/content/sections/cultureFit.js'
import { compensation } from '../../../src/content/sections/compensation.js'
import { contactMethods } from '../../../src/content/sections/contactMethods.js'

function strong(s: string): StrongSpan {
  return { kind: 'strong', children: [text(s)] }
}

function labeledBullet(label: string, description: string): ListItem {
  return {
    children: [paragraph([strong(label), text(' — '), text(description)] as readonly InlineSpan[])],
  }
}

export function extract(ctx: ExtractorContext): PageDoc {
  const body: DocNode[] = []

  // Hero subtitle as lead paragraph (hero.title lives in PageDoc.title via CONTACT_META, not body — D-11)
  body.push(paragraph(hero.subtitle))

  // roleFit
  body.push(heading(2, roleFit.heading))
  body.push(paragraph(roleFit.intro))
  body.push(paragraph(roleFit.lookingForHeading))
  body.push(list({
    ordered: false,
    items: roleFit.lookingFor.map((lf) => labeledBullet(lf.name, lf.description)),
  }))
  body.push(paragraph(roleFit.notLookingForHeading))
  body.push(list(roleFit.notLookingFor))

  // companyFit
  body.push(heading(2, companyFit.heading))
  body.push(list({
    ordered: false,
    items: companyFit.criteria.map((c) => labeledBullet(c.thesis, c.body)),
  }))

  // cultureFit
  body.push(heading(2, cultureFit.heading))
  body.push(list({
    ordered: false,
    items: cultureFit.items.map((i) => labeledBullet(i.lead, i.body)),
  }))

  // compensation
  body.push(heading(2, compensation.heading))
  body.push(list({
    ordered: false,
    items: compensation.rows.map((r) => labeledBullet(r.term, r.value)),
  }))
  // caption emitted as italicized paragraph after the rows
  body.push(paragraph([
    { kind: 'emphasis', children: [text(compensation.caption)] } as InlineSpan,
  ]))

  // contactMethods
  body.push(heading(2, contactMethods.heading))
  body.push(paragraph(contactMethods.intro))
  body.push(list({
    ordered: false,
    items: contactMethods.links.map((l) => ({
      children: [paragraph([
        strong(l.label),
        text(' — '),
        link(l.href, l.display),
      ] as readonly InlineSpan[])],
    })),
  }))
  body.push(paragraph(contactMethods.guidance))

  // colleagueQuotes — NO heading field in content/contact.ts, so no H2 (D-12)
  for (const q of colleagueQuotes) {
    const children: DocNode[] = [paragraph(q.quote)]
    if (q.cite !== undefined) {
      children.push(paragraph([{ kind: 'emphasis', children: [text(q.cite)] } as InlineSpan]))
    }
    if (q.context !== undefined) {
      children.push(paragraph([{ kind: 'emphasis', children: [text(q.context)] } as InlineSpan]))
    }
    body.push(blockquote(children))
  }

  return {
    ...CONTACT_META,
    sourceRoute: ctx.sourceRoute,
    body,
  }
}
```
