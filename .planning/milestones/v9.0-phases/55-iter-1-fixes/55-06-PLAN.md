export function caseFilesIndexTiddler(
  exhibits: readonly ExhibitJson[],
): Tiddler {
  const investigations: string[] = []

New contract — body layout (TiddlyWiki table wikitext):

```
All case files in a sortable table. Click a column header in TiddlyWiki to sort; filter by tag to narrow (tags: investigation-report, engineering-brief, {client}).

|!Date |!Client |!Type |!Case |
|2015 |General Dynamics |Investigation |[[A — Cross-Domain SCORM Resolution & Embedded Technical Advisory]] |
|2016 |Exelon |Brief |[[B — Leadership Recognition Chain]] |
```

Table wikitext rules (TW5 syntax):
- First row with `|!Header |!Header |...|` denotes column headers (the leading `!` after `|` marks a header cell).
- Subsequent rows use `|cell |cell |...|` (no `!`).
- Cells may contain arbitrary wikitext including `[[link]]`.
- Blank line after the table closes it.
- A one-sentence caption above the table explains what it is.

Type-cell mapping:
- `investigation-report` → `Investigation`
- `engineering-brief` → `Brief`
- any other value → pass through `ex.exhibitType` verbatim (defensive fallback)

Row construction:
```typescript
function typeCellFor(exhibitType: string): string {
  if (exhibitType === 'investigation-report') return 'Investigation'
  if (exhibitType === 'engineering-brief') return 'Brief'
  return exhibitType
}

function rowFor(ex: ExhibitJson): string {
  const case_ = `[[${ex.label} — ${ex.title}]]`
  return `|${ex.date} |${ex.client} |${typeCellFor(ex.exhibitType)} |${case_} |`
}

const sortedRows = [...exhibits]
  .slice()
  .sort((a, b) => (a.label < b.label ? -1 : a.label > b.label ? 1 : 0))
  .map(rowFor)

const text =
  `All case files in a sortable table. Click a column header in TiddlyWiki to sort; filter by tag to narrow (tags: investigation-report, engineering-brief, {client}).\n\n` +
  `|!Date |!Client |!Type |!Case |\n` +
  sortedRows.join('\n') +
  '\n'
```

export function caseFilesIndexTiddler(
  exhibits: readonly ExhibitJson[],
): Tiddler {
  const sorted = exhibits.slice().sort(compareByLabel)
  const rows: string[] = []
  for (const ex of sorted) {
    const caseLink = `[[${ex.label} — ${ex.title}]]`
    rows.push(`|${ex.date} |${ex.client} |${typeCellFor(ex.exhibitType)} |${caseLink} |`)
  }
  const text =
    `All case files in a sortable table. Click a column header in TiddlyWiki to sort; filter by tag to narrow (tags: investigation-report, engineering-brief, {client}).\n\n` +
    `|!Date |!Client |!Type |!Case |\n` +
    rows.join('\n') +
    '\n'
  return {
    title: 'Case Files Index',
    type: 'text/vnd.tiddlywiki',
    tags: ['page', 'case-files'],
    fields: {
      created: TIMESTAMP,
      modified: TIMESTAMP,
    },
    text,
  }
}
