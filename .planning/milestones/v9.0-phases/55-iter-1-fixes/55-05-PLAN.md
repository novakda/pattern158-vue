Rules for each footer section:
- `! Related questions` block: omit if the FAQ has no categories OR if no siblings exist in any of its categories. When multiple categories exist per FAQ, union the siblings (dedupe by question string). Exclude the FAQ itself. Sort alphabetically for determinism.
- `! Referenced exhibits` block: omit when the answer text contains zero exhibit-label references. Detection: scan answer for the regex `/\bExhibit\s+([A-Z])\b/g` AND only emit the link if that label exists in `ctx.exhibitLabels` (supplied from Plan 55-01 bundle.exhibits). Sort alphabetically. If ctx is undefined, skip this block entirely.
- `[[FAQ Index]]` backlink stays at the bottom of every footer (it is the only pre-existing link and the only one without a heading prefix).

// Per FAQ: gather siblings, dedupe, exclude self, sort
function siblingsFor(item: FaqJsonItem): string[] {
  const siblings = new Set<string>()
  for (const cat of item.categories) {
    const bucket = questionsByCategory.get(cat) ?? []
    for (const q of bucket) if (q !== item.question) siblings.add(q)
  }
  return Array.from(siblings).sort()
}

const EXHIBIT_REF_REGEX = /\bExhibit\s+([A-Z])\b/g

function exhibitReferencesIn(answer: string, known: readonly string[]): string[] {
  if (known.length === 0) return []
  const knownSet = new Set(known)
  const found = new Set<string>()
  for (const match of answer.matchAll(EXHIBIT_REF_REGEX)) {
    const label = match[1]
    if (knownSet.has(label)) found.add(label)
  }
  return Array.from(found).sort()
}

```typescript
export function faqItemsToTiddlers(
  items: readonly FaqJsonItem[],
  ctx?: FaqFooterContext,
): Tiddler[] {
  // Precompute: category → list of questions in that category
  const questionsByCategory = new Map<string, string[]>()
  for (const item of items) {
    for (const cat of item.categories) {
      const bucket = questionsByCategory.get(cat)
      if (bucket === undefined) questionsByCategory.set(cat, [item.question])
      else bucket.push(item.question)
    }
  }
  const knownExhibitLabels = ctx?.exhibitLabels ?? []
  const knownExhibitSet = new Set(knownExhibitLabels)

  return items.map((item) => {
    // Siblings
    const siblings = new Set<string>()
    for (const cat of item.categories) {
      const bucket = questionsByCategory.get(cat) ?? []
      for (const q of bucket) {
        if (q !== item.question) siblings.add(q)
      }
    }
    const sortedSiblings = Array.from(siblings).sort()

    // Exhibit refs
    const foundLabels = new Set<string>()
    if (knownExhibitSet.size > 0) {
      for (const match of item.answer.matchAll(EXHIBIT_REF_REGEX)) {
        const label = match[1]
        if (knownExhibitSet.has(label)) foundLabels.add(label)
      }
    }
    const sortedLabels = Array.from(foundLabels).sort()

    // Assemble footer
    const footerBlocks: string[] = []
    if (sortedSiblings.length > 0) {
      const lines = sortedSiblings.map((q) => `* [[${q}]]`).join('\n')
      footerBlocks.push(`! Related questions\n\n${lines}`)
    }
    if (sortedLabels.length > 0) {
      const lines = sortedLabels.map((l) => `* [[Exhibit ${l}]]`).join('\n')
      footerBlocks.push(`! Referenced exhibits\n\n${lines}`)
    }
    const footerBody = footerBlocks.length > 0
      ? `${footerBlocks.join('\n\n')}\n\n`
      : ''

    const body =
      `${renderAnswerParagraphs(item.answer)}\n\n` +
      `---\n\n` +
      `!! See also\n\n` +
      `${footerBody}[[FAQ Index]]\n`

    return {
      title: item.question,
      type: 'text/vnd.tiddlywiki',
      tags: ['faq', ...item.categories],
      fields: {
        created: TIMESTAMP,
        modified: TIMESTAMP,
        'faq-id': item.id,
      },
      text: body,
    }
  })
}
```
