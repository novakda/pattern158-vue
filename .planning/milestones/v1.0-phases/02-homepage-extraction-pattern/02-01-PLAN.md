**TechTags.vue — extend props to accept `(Tag | string)[]`:**

Replace the entire `<script setup lang="ts">` block with:

```typescript
import { computed } from 'vue'

export interface Tag {
  label: string
  title: string
}

const props = defineProps<{
  tags: (Tag | string)[]
}>()

const normalizedTags = computed<Tag[]>(() =>
  props.tags.map(t => typeof t === 'string' ? { label: t, title: t } : t)
)
```

Replace the template to use `normalizedTags` instead of `tags`:

```html
<template>
  <div class="tech-tags">
    <span v-for="tag in normalizedTags" :key="tag.label" class="impact-tag" :title="tag.title">{{ tag.label }}</span>
  </div>
</template>
```
